void drawDisplay(const ClockTime& now) {
  char timeText[16];
  formatTimeString(now, timeText, sizeof(timeText));

  display.clearBuffer();
  display.setFont(u8g2_font_ncenB14_tr);
  display.drawStr(0, 16, timeText);

  display.setFont(u8g2_font_6x12_tr);
  drawBinaryRow( H, now.hour, 6, 32);
  drawBinaryRow(M, now.minute, 6, 46);
  drawBinaryRow(S, now.second, 6, 60);

  display.sendBuffer();
}
