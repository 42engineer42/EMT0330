void setup() {
  Serial.begin(115200);
  while (!Serial) {
    delay(10);
  }
  if (!display.begin(SSD1306_SWITCHCAPVCC, kOledAddress)) {
    Serial.println(F( SSD1306 allocation failed));
    while (true) {
      delay(1000);
    }
  }
  display.clearDisplay();
  display.setTextColor(SSD1306_WHITE);
  referenceMillis = millis();
  printPrompt();
}
